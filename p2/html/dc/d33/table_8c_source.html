429 PN potion_tuple_bsearch(Potion *P, PN cl, PN self, PN x) {
430  struct PNTuple *t = PN_GET_TUPLE(self);
431  PNUniq xu = PN_UNIQ(x);

433  while (i <= j) {

435  PNUniq u = PN_UNIQ(t->set[m]);

447 static void potion_sort_internal(Potion *P, PN cl, PN self,
448  PN_SIZE from,
449  PN_SIZE to,
450  PN cmp)
451 {
452  if (from < to) {
453  struct PNTuple *t = PN_GET_TUPLE(self);
454  // which pivot? first is worst case if already sorted.
455  // random, last, middle or best: median-of-3.
456  PN_SIZE i, index = from + (to - from)/2;
457  PN pivot = GET(index);
458  SWAP(index, to);
459  index = from;
460  // partition the portion of the tuple between indexes from and to,
461  // inclusively, by moving all elements less than pivot before
462  // the pivot, and the equal or greater elements after it.
463  if (cmp == PN_NIL) { // default: sort by uniq, not value
464  for (i=from; i < to-1; i++) { // from i < to
465  if (PN_UNIQ(GET(i)) <= PN_UNIQ(pivot)) { SWAP(i, index); index++; }
466  }
467  } else if (cmp == PN_TRUE) { // sort by ascending number
468  for (i=from; i < to; i++) {
469  if (GET(i) <= pivot) {
470  SWAP(i, index); index++;
471  }
472  }
473  } else if (cmp == PN_FALSE) { // sort by descending number
474  for (i=from; i < to; i++) {
475  if (GET(i) > pivot) { SWAP(i, index); index++;
476  }
477  }
478  } else {
479  vPN(Closure) c = PN_CLOSURE(cmp);
480  for (i=from; i < to; i++) { // call cmp
481  if (PN_INT(c->method(P, cl, cmp, GET(i), pivot)) > 0)
482  { SWAP(i, index); index++; }
483  }
484  }
485  SWAP(index, to); // Move pivot element back to its final place
486 
487  if (index > 0)
488  potion_sort_internal(P,cl,self, from, index-1, cmp);
489  potion_sort_internal(P,cl,self, index+1, to, cmp);
490  PN_TOUCH(self);
491  }
492 }

525 PN potion_tuple_ins_sort(Potion *P, PN cl, PN self, PN cmp) {
526  struct PNTuple *t = PN_GET_TUPLE(self);
527  unsigned long i, j;
528  vPN(Closure) c;
529  if (t->len < MAX_INS_SORT) {
530  // simple insertion sort for smaller arrays (<13)
531  if (cmp == PN_NIL) { // default: sort by uniq, not value
532  for (i = 1; i < t->len; i++) {
533  j = i;
534  while (j > 0 && PN_UNIQ(GET(j-1)) > PN_UNIQ(GET(j))) {
535  SWAP(j, j-1);
536  j--;
537  }
538  }
539  }
540  else if (PN_IS_CLOSURE(cmp)) {
541  c = PN_CLOSURE(cmp);
542  for (i = 1; i < t->len; i++) {
543  j = i;
544  while (j > 0 && PN_INT(c->method(P, cl, cmp, GET(j-1), GET(j))) > 0) {
545  SWAP(j, j-1);
546  j--;
547  }
548  }
549  }
550  else if (cmp == PN_TRUE) {
551  for (i = 1; i < t->len; i++) {
552  j = i;
553  while (j > 0 && GET(j-1) > GET(j)) {
554  SWAP(j, j-1);
555  j--;
556  }
557  }
558  }
559  else if (cmp == PN_FALSE) {
560  for (i = 1; i < t->len; i++) {
561  j = i;
562  while (j > 0 && GET(j-1) < GET(j)) {
563  SWAP(j, j-1);
564  j--;
565  }
566  }
567  }
568  else {
569  potion_fatal("sort: invalid cmp type");
570  }
571  PN_TOUCH(self);
572  }
573  else {
574  if (cmp != PN_NIL && !PN_IS_BOOL(cmp) && !PN_IS_CLOSURE(cmp))
575  potion_fatal("sort: invalid cmp type");
576  potion_sort_internal(P, cl, self, 0, t->len-1, cmp);
577  }
578  return self;
579 }

581 static PN potion_tuple_cmp(Potion *P, PN cl, PN self, PN value) {
582  switch (potion_type(value)) {
583  case PN_TBOOLEAN: // false < () < true
584  return value == PN_FALSE ? -1 : 1;
585  case PN_TNIL:
586  return -1; //nil < () < (...)
587  case PN_TTUPLE: // recurse
588  if(PN_TUPLE_LEN(self) && PN_TUPLE_LEN(value)) {
589  PN cmp;
590  if ((cmp = potion_send(potion_tuple_first(P,cl,self), PN_cmp,
591  potion_tuple_first(P,cl,value)))
592  == PN_NUM(0))
593  {
594  PN t1 = potion_tuple_clone(P,cl,self);
595  PN t2 = potion_tuple_clone(P,cl,value);
596  potion_tuple_pop(P,cl,t1);
597  potion_tuple_pop(P,cl,t2);
598  return potion_send(t1, PN_cmp, t2);
599  }
600  else {
601  return cmp;
602  }
603  }
604  else {
605  if (PN_TUPLE_LEN(value)) return -1;
606  else if (PN_TUPLE_LEN(self)) return 1;
607  else return 0;
608  }
609  default:
610  potion_fatal("Invalid tuple cmp type");
611  }
612 }
