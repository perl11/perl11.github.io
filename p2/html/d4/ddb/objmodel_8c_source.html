129 PN potion_class(Potion *P, PN cl, PN self, PN ivars) {
130  PN parent = (self == P->lobby ? PN_VTABLE(PN_TOBJECT) : self);
131  PN pvars = ((struct PNVtable *)parent)->ivars;
132  PNType t = PN_FLEX_SIZE(P->vts) + PN_TNIL;
133  PN_FLEX_NEEDS(1, P->vts, PN_TFLEX, PNFlex, TYPE_BATCH_SIZE);
134  PN_FLEX_SIZE(P->vts)++;
135  self = potion_type_new(P, t, parent);
136  if (PN_IS_TUPLE(pvars)) {
137  if (!PN_IS_TUPLE(ivars)) ivars = PN_TUP0();
138  PN_TUPLE_EACH(pvars, i, v, {PN_PUT(ivars, v);});
139  }
140  if (PN_IS_TUPLE(ivars))
141  potion_ivars(P, PN_NIL, self, ivars);
142 
143  if (!PN_IS_CLOSURE(cl))
144  cl = ((struct PNVtable *)parent)->ctor;
145  ((struct PNVtable *)self)->ctor = cl;
146 
147  PN_TOUCH(P->vts);
148  return self;
149 }

215 PN potion_def_method(Potion *P, PN closure, PN self, PN key, PN method) {
216  int ret;
217  PN cl;
218  vPN(Vtable) vt = (struct PNVtable *)self;
219  unsigned k = kh_put(PN, vt->methods, key, &ret);
220  PN_QUICK_FWD(struct PNTable *, vt->methods);
221  PN_TOUCH(vt->methods);
222 
223  if (!PN_IS_CLOSURE(method)) {
224  if (PN_IS_PROTO(method))
225  cl = potion_closure_new(P, (PN_F)potion_proto_method, PN_PROTO(method)->sig, 1);
226  else
227  cl = potion_closure_new(P, (PN_F)potion_getter_method, PN_NIL, 1);
228  PN_CLOSURE(cl)->data[0] = method;
229  method = cl;
230  }
231 
232  kh_val(PN, vt->methods, k) = method;
233  PN_TOUCH(self);
234 
235 #ifdef JIT_MCACHE
236  // TODO: make JIT_MCACHE more flexible, store in fixed gc, see ivfunc TODO also
237  // TODO: this is disabled until method weakrefs can be stored in fixed memory
238  if (P->target.mcache != NULL) {
239  PNAsm * volatile asmb = potion_asm_new(P);
240  P->target.mcache(P, vt, &asmb);
241  if (asmb->len <= 4096) {
242  if (vt->mcache == NULL)
243  vt->mcache = PN_ALLOC_FUNC(4096);
244  PN_MEMCPY_N(vt->mcache, asmb->ptr, u8, asmb->len);
245  } else if (vt->mcache != NULL) {
246  potion_munmap(vt->mcache, 4096);
247  vt->mcache = NULL;
248  }
249  }
250 #endif
251  return method;
252 }

265 PN potion_bind(Potion *P, PN rcv, PN msg) {
266  PN closure = PN_NIL;
267  PN vt = PN_NIL;
268  PNType t = PN_TYPE(rcv);
269  if (!PN_TYPECHECK(t)) return PN_NIL;
270  if (t == PN_TVTABLE && !PN_IS_METACLASS(rcv)) {
271  vt = (PN)((struct PNVtable *)rcv)->meta;
272  } else {
273  vt = PN_VTABLE(t);
274  }
275  while (1) {
276  closure = ((msg == PN_lookup) && (t == PN_TVTABLE))
277  ? potion_lookup(P, 0, vt, msg)
278  : potion_send(vt, PN_lookup, msg);
279  if (closure || !((struct PNVtable *)vt)->parent) break;
280  vt = PN_VTABLE(((struct PNVtable *)vt)->parent);
281  }
282  return closure;
283 }
